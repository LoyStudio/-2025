      var container = document.querySelector("#unity-container");
      var canvas = document.querySelector("#unity-canvas");

      var buildUrl = "Build/YourGameName";
      var loaderUrl = buildUrl + ".loader.js";
      var config = {
        dataUrl: buildUrl + ".data",
        frameworkUrl: buildUrl + ".framework.js",
        codeUrl: buildUrl + ".wasm",
        streamingAssetsUrl: "StreamingAssets",
        companyName: "YourCompanyName",
        productName: "YourGameName",
        productVersion: "0.1",
      };

      var script = document.createElement("script");
      script.src = loaderUrl;
      script.onload = () => {
        createUnityInstance(canvas, config).then((unityInstance) => {
          // Игра запущена
        });
      }
      document.body.appendChild(script);

      // === Адаптация размера Canvas ===
      function resizeCanvas() {
        var gameWidth = 9; // Ширина (9:16)
        var gameHeight = 16; // Высота (9:16)
        var windowWidth = window.innerWidth;
        var windowHeight = window.innerHeight;

        var targetWidth = windowWidth;
        var targetHeight = windowWidth * gameHeight / gameWidth; // Вычисляем высоту, сохраняя пропорции

        if (targetHeight > windowHeight) {
          // Если вычисленная высота больше высоты окна, используем высоту окна
          targetHeight = windowHeight;
          targetWidth = windowHeight * gameWidth / gameHeight; // Вычисляем ширину
        }

        canvas.width = targetWidth;
        canvas.height = targetHeight;

        // Применяем стили для центрирования и полос
        container.style.width = windowWidth + 'px';
        container.style.height = windowHeight + 'px';
        container.style.display = 'flex';
        container.style.justifyContent = 'center';
        container.style.alignItems = 'center';
      }

      window.addEventListener('resize', resizeCanvas);
      resizeCanvas(); // Вызвать при загрузке страницы
    </script>
